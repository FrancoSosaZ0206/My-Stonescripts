// VARIABLES

var runCount = 0
var keepMask = true

var smiteScr = 1
var weavScr = 4

var canDebI
var canDebP
var canDebF

var keepFTali = false

?p.f = "boss"
  keepFTali = true

var fBowCount = 0

?p.ll
  keepMask = false
  keepFTali = false
  fBowCount = 0

  ?p.tt = 1
    runCount++

    ?p.ls > 5 & runCount >= 2 & smiteScr ! 3
      smiteScr++



// FUNCTIONS

func ldtF(mode) // loadouts

  ?mode = "default"
    equipL triskelion
    ?!keepMask
      equipR @a.cshield1@
  :?mode = "armorRegen"
    equipL triskelion
    ?!keepMask
      equipR @a.cshield2@
  :?mode = "engage"
    equipL triskelion
    ?!keepMask
      equipR @a.aeshield@
  :?mode = "escape"
    equipL mind
    ?!keepMask
      equipR @a.aeshield@
  :?mode = "magnet"
    equipL star
    ?!keepMask
      equipR triskelion

  :?mode = "extend"
    equipL triskelion
    equipR mask

  :?mode = "unmkDmg"
    ?p.tt % 2 = 0 & p.fd > 4
      equipL triskelion
      equipR hatchet
    :
      equip @a.dUstaff@

  :?mode = "rangedDmg"
    equip @a.aebow@
  :?mode = "rangedP"
    equip @a.dPbow@
  :?mode = "rangedF"
    equip @a.fbow@

  :?mode = "magic"
    ?keepMask
      ?canDebI
        equipL @a.dIwand1@
      :?canDebP
        equipL @a.dPwand@
      :?canDebF
        equipL @a.dFwand@

    :
      ?p.tt % 2 = 0
        equipL triskelion
        equipR hatchet

      :?canDebI
        equip @a.dIstaff@
      :?canDebP
        equip @a.dPstaff@
      :?canDebF
        equip @a.dFstaff@

  :
    >`0,2,#red,ERROR: ldtF() recieved incorrect mode: @mode@

  return

func fight1v1()
  p.fds = foe.debuffs.string
  p.php = hp
  p.pmhp = maxhp

  /* How to apply debuffs:
  - apply all except the ones the foe is immune to.
  Finish when:
  - all possible debuffs have been applied. */
  canDebI = (p.fds ! "chill:6" & p.f ! "immune_to_debuff_chill")
  canDebP = (p.fds ! "damage" & p.f ! "immune_to_debuff_damage")
  canDebF = (p.fds ! "dot" & p.f ! "immune_to_debuff_dot")

  ?canDebI | canDebP | canDebF
    ?keepMask
      ?canDebI
        equipL @a.dIsword@
      :?canDebP
        equipL @a.dPsword1@
      :?canDebF
        equipL @a.dFsword@
    :
      //Left hand:
      ?canDebP
        equipL @a.dPsword1@
      :?canDebF
        equipL @a.dFsword@
      :?canDebI
        equipL @a.dIwand1@

      //Right hand:
      ?canDebI
        equipR @a.dIsword@

  :?p.php ! p.pmhp
    ?keepMask
      equipL @a.dLsword@
    :
      equipL moon
      equipR @a.dLsword@

  :
    ?keepMask
      equipL Void Slayer
    :
      p.bs = buffs.string
      ?p.bs = "berserk"
        c.doMoondial(Void Slayer, a.aesword, 2)
      :
        c.doMoondial(Void Slayer, a.aesword, 0)

  return



// PROCEDURE:


c.doAAC("sword 0*", a.cshield1)

/*?p.lb // AUTO BREWING:

?keepMask
  ?p.ir ! "bardiche" &
  ^p.ir ! "quarterstaff" &
  ^p.ir ! "heavy hammer"
    equipR mask

?p.lb | p.ll
  equipL triskelion
  equipR @a.cshield1@

:?c.canUseAETali() & !keepFTali
  ?keepMask
    c.useAETali("l")
  :
    equipL triskelion
    c.useAETali("r")

:?c.canUseFTali() & keepFTali
  ?keepMask
    c.useFTali("l")
  :
    equipL triskelion
    c.useFTali("r")

:?c.canUsePot() & keepMask
  activate potion

:?pickup.distance < 15
  ldtF("magnet")
:
  p.fd = foe.distance
  p.aiE = ai.enabled

  ?c.canUseBladeDS(smiteScr)
    c.useBladeDS()

  /****************************/
  ?c.canUseWeav()
    ?p.si = weavScr & p.fd < 15
    ^& keepMask
      c.useWeav()
  // **************************/

  ?p.fd > 35
    p.pa = armor

    ?c.canUseQstaff() & p.aiE
    ^& p.ir ! "bardiche" & p.ir ! Kubikiribocho
    ^& p.ir ! "heavy hammer"
      c.useQstaff(a.myqstaff)

    :?14 <= p.pa & p.pa <= 29
      ldtF("default")
    :?p.pa < 14
      ldtF("armorRegen")

    :
      equipL triskelion
      equipR mask
  :
    p.fds = foe.debuffs.string

    ?p.f ! "bronze_guardian"
      ?p.fd > 15
        ?c.canUseQstaff() & p.aiE
        ^& p.ir ! "bardiche" &
        ^  p.ir ! Kubikiribocho &
        ^  p.ir ! "heavy hammer" &
        ^(p.f ! "explode" & p.fd > 17)
          c.useQstaff(a.myqstaff)

        :
          ldtF("engage")

          var isAbtToDash
          isAbtToDash = (p.fd <= 17)
          ?keepMask & isAbtToDash
            equipR @a.aeshield@

          canDebI = (p.fds ! "chill" &
          ^f ! "immune_to_debuff_chill")
          p.bs = buffs.string

          ?!keepMask & canDebI & isAbtToDash &
          ^i.getFoeTotalHP() > i.getBuffInfo("smite", false, 2) * a.smiteMult
            equipR @a.aiShield@

      :?c.canBash()
        c.bash(Void Slayer)
      :?c.canDash()
        c.dash(Void Slayer,a.dshield)

      :?p.f = "explode" // minecart miniboss
        ?armor < 80 // we don't need to escape
          ldtF("escape") // if we can tank the damage.

      :?c.canUseQstaff() & p.aiE
      ^& p.ir ! "bardiche"
      ^& p.ir ! Kubikiribocho
      ^& p.ir ! "heavy hammer"
        c.useQstaff(a.myqstaff)

      :
        ?c.canUseHamm() & keepMask
        ^& p.fd < 15 &
        ^((smiteScr = 1 & p.si = 3) |
        ^ smiteScr = 3 & p.si = 2    )
          c.useHamm(a.myHamm)

        :?p.fc = 1
          fight1v1()

        :
          canDebI = (p.fds ! "chill:6" & p.f ! "immune_to_debuff_chill")
          canDebP = (p.fds ! "damage" & p.f ! "immune_to_debuff_damage")
          canDebF = (p.fds ! "dot" & p.f ! "immune_to_debuff_dot")

          ?canDebI | canDebP | canDebF
            ldtF("magic")
          :
            ldtF("unmkDmg")


    : // p.f = "bronze_guardian"
      p.fs = foe.state
      p.ft = foe.time

      ?p.fs = 4 // dying
        ldtF("magnet")

      :?p.fs ! 1 // boss appearing cinematic
        ?c.canUseCind() & keepMask
          ?p.fds = a.maxIgnitions_Default |
          ^c.canKillCind(i.getFoeTotalHP(),
          ^              i.getDebuffLvl("ignition", true),
          ^              a.cindDmg_Default, dpsm.DPS)
            c.useCind()

        ?c.canUsePotDmg() |
        ^c.canUsePotSpecial("strength",0)
          activate potion
        :?c.canUseHamm() & keepMask
          c.useHamm(a.myHamm)
        :?c.canUseMask() & keepMask
          c.useMask()

        :?p.fs = 32 & p.ft >= u.sec // boss's hammer attack
          ldtF("escape")

        :?p.fs = 33 & p.ft < 6*u.sec // boss's hammer's stuck

          ?p.ft > 5 & c.canUseQstaff()
            c.useQstaff(a.myqstaff)

          :?p.fd > 15
            equipL triskelion // get close fast to boss
            ?!keepMask
              equipR Void Slayer
          :?c.canBash()
            c.bash(Void Slayer)
          :?c.canDash()
            dash(VoidSlayer,a.dshield)

          :
            ?c.canUseBard() & keepMask
              c.useBard(Kubikiribocho)
            :
              fight1v1()

        :
          ?p.irs = 3
            fBowCount++
          ?fBowCount = 0
            ldtF("rangedF")

          :
            ?p.fa < 1
              ?c.canUseMask() & !keepMask
                c.useMask()
              :?c.canUseHamm() & !keepMask & p.ip = "strength"
                c.useHamm(a.myHamm)
              :
                var pBowCount = fBowCount + 1
                ?p.fds ! "damage" & pBowCount ! fBowCount
                ^& p.ip = "strength"
                  ldtF("rangedP")

                :?keepMask
                  equipL @a.altbow@
                :
                  ldtF("rangedDmg")

            :?keepMask
              equipL @a.altbow@
            :
              ldtF("rangedDmg")
